# リサーチ発掘（プランB＋C） 実装計画

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 既存 `@yancha/research` に、キーワードによる競合チャンネル自動発掘（B）と、候補の人間承認・出自タグ・退役・クォータガードという統合運用（C）を追加する。

**Architecture:** プランAの `youtube`/`ledger`/`collect` を再利用し、`keywords.yaml → discover(search.list＋統計肉付け＋フィルタ) → candidates.yaml → approve/reject → ledger.yaml` の流れを足す。発掘は `search.list`（100ユニット/回）、統計肉付けは既存 `channels.list`（1ユニット系）。候補は自動投入せず人間が承認する。常駐なし・コマンド駆動。

**Tech Stack:** 既存 `@yancha/research`（TypeScript ESM / tsx / vitest / yaml / better-sqlite3 / Hono）に追加。新パッケージは作らない。

## Global Constraints

- 対象パッケージ: `packages/research`（既存）。新規パッケージを作らない。
- 言語: TypeScript ESM、import は拡張子 `.js` を明示。コメント/ログ/エラーは日本語。
- テスト: vitest（globals 有効）。外部API呼び出しは `fetchImpl` 差し替えでモックする。
- エラーは `@yancha/core` の `YanchaError`（code 付き）で送出。
- **発掘は `search.list`（`part=snippet`, `type=channel`）= 100ユニット/回。1キーワード1回、ページングしない。**
- **候補を台帳へ自動投入しない。** `candidates.yaml` に貯め、`approve` で人間が昇格する。
- **自動退役しない。** `retire` は人間が実行し `retiredAt` を付けるだけ（行は消さない）。
- 常駐・cron・Docker を作らない。すべてコマンド駆動。
- コミットは日本語 `prefix: 要約`。関連Issueは `#16`（B）/ `#18`（C）。

import { readFile } from "node:fs/promises";
import { parse } from "yaml";
import type { KeywordsFile, KeywordsSettings } from "./types.js";

export const DEFAULT_KEYWORDS_SETTINGS: KeywordsSettings = {
  minSubscribers: 100,
  maxCandidatesPerRun: 30,
  relevanceLanguage: "ja",
  regionCode: "JP",
  quotaGuardUnits: 3000
};

// search.list は1回100ユニット固定
export function estimateSearchUnits(keywordCount: number): number {
  return keywordCount * 100;
}

export async function readKeywordsFile(path: string): Promise<KeywordsFile> {
  let raw: string;
  try {
    raw = await readFile(path, "utf8");
  } catch (error) {
    if ((error as NodeJS.ErrnoException).code === "ENOENT") {
      return { keywords: [], settings: DEFAULT_KEYWORDS_SETTINGS };
    }
    throw error;
  }

  const parsed = parse(raw) as { keywords?: string[]; settings?: Partial<KeywordsSettings> } | null;
  return {
    keywords: parsed?.keywords ?? [],
    settings: { ...DEFAULT_KEYWORDS_SETTINGS, ...(parsed?.settings ?? {}) }
  };
}
```

import { mkdir, readFile, writeFile } from "node:fs/promises";
import { dirname } from "node:path";
import { parse, stringify } from "yaml";
import type { Candidate } from "./types.js";

export async function readCandidates(path: string): Promise<Candidate[]> {
  let raw: string;
  try {
    raw = await readFile(path, "utf8");
  } catch (error) {
    if ((error as NodeJS.ErrnoException).code === "ENOENT") {
      return [];
    }
    throw error;
  }
  const parsed = parse(raw) as { candidates?: Candidate[] } | null;
  return parsed?.candidates ?? [];
}

async function writeCandidates(path: string, candidates: readonly Candidate[]): Promise<void> {
  await mkdir(dirname(path), { recursive: true });
  await writeFile(path, stringify({ candidates }), "utf8");
}

export async function addCandidates(path: string, incoming: readonly Candidate[]): Promise<Candidate[]> {
  const current = await readCandidates(path);
  const known = new Set(current.map((c) => c.channelId));
  const merged = [...current];
  for (const candidate of incoming) {
    if (!known.has(candidate.channelId)) {
      known.add(candidate.channelId);
      merged.push(candidate);
    }
  }
  await writeCandidates(path, merged);
  return merged;
}

export async function removeCandidate(path: string, channelId: string): Promise<Candidate[]> {
  const current = await readCandidates(path);
  const remaining = current.filter((c) => c.channelId !== channelId);
  await writeCandidates(path, remaining);
  return remaining;
}
```

```typescript
import { YanchaError } from "@yancha/core";
import { estimateSearchUnits } from "./keywords.js";
import type { YoutubeClient } from "./youtube.js";
import type { Candidate, KeywordsSettings } from "./types.js";

export interface DiscoverDeps {
  readonly keywords: readonly string[];
  readonly settings: KeywordsSettings;
  readonly client: Pick<YoutubeClient, "searchChannels" | "fetchChannels">;
  readonly existingChannelIds: ReadonlySet<string>;
  readonly confirmed: boolean;
}

// 配列をsize件ずつに分割
function chunk<T>(items: readonly T[], size: number): T[][] {
  const chunks: T[][] = [];
  for (let i = 0; i < items.length; i += size) {
    chunks.push(items.slice(i, i + size));
  }
  return chunks;
}

export async function discoverCandidates(deps: DiscoverDeps): Promise<Candidate[]> {
  // クォータガード: search前に見積りを検査
  const estimated = estimateSearchUnits(deps.keywords.length);
  if (estimated > deps.settings.quotaGuardUnits && !deps.confirmed) {
    throw new YanchaError(
      "POLICY_VIOLATION",
      `クォータ見積り ${estimated} ユニットが上限 ${deps.settings.quotaGuardUnits} を超えます。--yes で続行してください。`
    );
  }

  const discoveredAt = new Date().toISOString();

  // 各キーワードで検索し、channelId→一致キーワードを集約
  const matched = new Map<string, Set<string>>();
  const titles = new Map<string, string>();
  for (const keyword of deps.keywords) {
    const results = await deps.client.searchChannels(keyword, {
      relevanceLanguage: deps.settings.relevanceLanguage,
      regionCode: deps.settings.regionCode
    });
    for (const result of results) {
      if (deps.existingChannelIds.has(result.channelId)) {
        continue; // 既存(台帳/候補)は除外
      }
      if (!matched.has(result.channelId)) {
        matched.set(result.channelId, new Set());
        titles.set(result.channelId, result.title);
      }
      matched.get(result.channelId)?.add(keyword);
    }
  }

  const uniqueIds = [...matched.keys()];
  if (uniqueIds.length === 0) {
    return [];
  }

  // 統計肉付け(channels.list=1ユニット系)を50件ずつ
  const subscriberById = new Map<string, { subscriberCount: number; title: string }>();
  for (const ids of chunk(uniqueIds, 50)) {
    const channels = await deps.client.fetchChannels(ids);
    for (const channel of channels) {
      subscriberById.set(channel.channelId, { subscriberCount: channel.subscriberCount, title: channel.title });
    }
  }

  // 登録者下限で絞り、降順に上限件数
  const candidates: Candidate[] = uniqueIds
    .map((channelId): Candidate => {
      const stat = subscriberById.get(channelId);
      return {
        channelId,
        title: stat?.title ?? titles.get(channelId) ?? "",
        subscriberCount: stat?.subscriberCount ?? 0,
        matchedKeywords: [...(matched.get(channelId) ?? [])],
        discoveredAt
      };
    })
    .filter((candidate) => candidate.subscriberCount >= deps.settings.minSubscribers)
    .sort((a, b) => b.subscriberCount - a.subscriberCount)
    .slice(0, deps.settings.maxCandidatesPerRun);

  return candidates;
}
```

// 退役: 行は消さずretiredAtを付ける（履歴保持）。collect対象からは除外される
export async function retireLedgerEntry(ledgerYamlPath: string, channelId: string): Promise<LedgerEntry[]> {
  const current = await readLedger(ledgerYamlPath);
  if (!current.some((entry) => entry.channelId === channelId)) {
    throw new YanchaError("CONFIG_INVALID", `台帳に存在しないchannelIdです: ${channelId}`);
  }
  const next = current.map((entry) =>
    entry.channelId === channelId && !entry.retiredAt ? { ...entry, retiredAt: new Date().toISOString() } : entry
  );
  await mkdir(dirname(ledgerYamlPath), { recursive: true });
  await writeFile(ledgerYamlPath, stringify({ channels: next }), "utf8");
  return next;
}

// 退役していないエントリだけを返す純関数
export function activeLedgerEntries(entries: readonly LedgerEntry[]): LedgerEntry[] {
  return entries.filter((entry) => !entry.retiredAt);
}
